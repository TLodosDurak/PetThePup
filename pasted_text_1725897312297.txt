const canvas = document.getElementById('yourCanvasId');

// Prevent default behavior on touch gestures and scroll
canvas.addEventListener('gesturestart', (e) => e.preventDefault());
canvas.addEventListener('gesturechange', (e) => e.preventDefault());
canvas.addEventListener('gestureend', (e) => e.preventDefault());

canvas.addEventListener('touchmove', (e) => {
  if (e.touches.length > 1) {
    e.preventDefault(); // Prevent multi-finger gestures like zoom
  }
}, { passive: false });  // passive: false ensures preventDefault works on touch events

// Optionally, prevent the scroll wheel zooming or other scroll events on the canvas
canvas.addEventListener('wheel', (e) => {
  e.preventDefault(); // Prevent scroll/zoom on wheel
}, { passive: false });

Key Points:
The e.preventDefault() method stops the default behavior of the event, which for gestures is usually zooming, switching apps, or other system-level actions.
Setting { passive: false } ensures that the preventDefault() call works on touch and wheel events.
This approach will prevent the zooming, switching between apps, or other multi-finger gestures while the user interacts with your Three.js canvas element.
